@override
Future<void> sendMainModeCommand(
  ControllerMode mode, {
  String? ssid,
  String? password,
}) async {
  if (!_canSend) return;

  final payload = jsonEncode({
    'type': 'set_main_mode',
    'mode': controllerModeToPayload(mode),
    'ssid': ?ssid,
    'pass': ?password,
  });

  await _writePayloadAwaited(
    payload,
    withoutResponse: false,
    errorContext: 'main mode command',
  );
}
